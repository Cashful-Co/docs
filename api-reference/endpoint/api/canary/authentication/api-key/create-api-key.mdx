Create a new API key for the current authenticated user/organization.

## Request Body

<ParamField body="name" type="string" required>
  A human-friendly name for the key (e.g., `My API Key`)
</ParamField>

<ParamField body="expiresIn" type="number">
  Expiration time in seconds (optional)
</ParamField>

<ParamField body="prefix" type="string">
  Prefix to use for the generated key (optional)
</ParamField>

<ParamField body="metadata" type="object">
  Optional metadata for the key
</ParamField>

## Responses

<ResponseField status="200" name="API key created successfully">
  <ResponseField name="id" type="string">ID of the created key</ResponseField>
  <ResponseField name="secret" type="string">The secret API key value (shown once)</ResponseField>
  <ResponseField name="prefix" type="string">Key prefix for display</ResponseField>
  <ResponseField name="name" type="string">Human-friendly name</ResponseField>
  <ResponseField name="expiresAt" type="string">ISO 8601 expiration timestamp (if set)</ResponseField>
</ResponseField>

<ResponseField status="400" name="Bad Request">
  Invalid input or validation error
</ResponseField>

<ResponseField status="401" name="Unauthorized">
  Invalid or missing authentication token
</ResponseField>

<ResponseField status="500" name="Internal Server Error">
  An unexpected error occurred on the server
</ResponseField>

## Authentication

This endpoint requires Bearer token authentication.
