Create a new user account using an email address and a password.

## Request Body

<ParamField body="email" type="string" required>
  User's email address
</ParamField>

<ParamField body="password" type="string" required>
  User's password
</ParamField>

<ParamField body="name" type="string">
  Optional full name of the user
</ParamField>

## Responses

<ResponseField status="200" name="User created successfully">
  <ResponseField name="userId" type="string">Created user's id</ResponseField>
  <ResponseField name="email" type="string">Created user's email</ResponseField>
</ResponseField>

<ResponseField status="400" name="Bad Request">
  Invalid input or validation error
</ResponseField>

<ResponseField status="500" name="Internal Server Error">
  An unexpected error occurred
</ResponseField>

## Example Request

```json
{
  "email": "jane@example.com",
  "password": "supersecret",
  "name": "Jane Doe"
}
```

## Example Response

```json
{
  "userId": "user_123",
  "email": "jane@example.com"
}
```
