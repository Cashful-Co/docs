Retrieves a log of all API events for debugging and logging.

## Query Parameters

<ParamField query="type" type="string">
  Filter by event type (e.g., "checkout.session.completed", "payment.succeeded")
</ParamField>

<ParamField query="startDate" type="string">
  Filter events from this date (ISO 8601 format)
</ParamField>

<ParamField query="endDate" type="string">
  Filter events until this date (ISO 8601 format)
</ParamField>

<ParamField query="offset" type="number">
  Number of events to skip (default: 0)
</ParamField>

<ParamField query="limit" type="number">
  Maximum number of events to return (default: 50)
</ParamField>

## Response

<ResponseField status="200" name="Success">
  <ResponseField name="data" type="array">
    Array of events
    <Expandable title="Event Object">
      <ResponseField name="id" type="string">
        Unique identifier for the event
      </ResponseField>
      <ResponseField name="merchantId" type="string">
        The merchant's unique identifier
      </ResponseField>
      <ResponseField name="type" type="string">
        Event type (e.g., "checkout.session.completed")
      </ResponseField>
      <ResponseField name="data" type="object">
        Event payload containing event-specific data
      </ResponseField>
      <ResponseField name="relatedEntityId" type="string">
        ID of the related entity triggering the event
      </ResponseField>
      <ResponseField name="relatedEntityType" type="string">
        Type of related entity (e.g., CheckoutSession)
      </ResponseField>
      <ResponseField name="createdAt" type="string">
        ISO 8601 timestamp when the event occurred
      </ResponseField>
    </Expandable>
  </ResponseField>
  <ResponseField name="pagination" type="object">
    Pagination information
    <Expandable title="Pagination">
      <ResponseField name="limit" type="number">
        Number of records returned
      </ResponseField>
      <ResponseField name="offset" type="number">
        Number of records skipped
      </ResponseField>
      <ResponseField name="total" type="number">
        Total number of records available
      </ResponseField>
      <ResponseField name="hasMore" type="boolean">
        Whether more records are available
      </ResponseField>
    </Expandable>
  </ResponseField>
</ResponseField>

## Common Event Types

<Card title="checkout.session.completed" icon="circle-check">
  Fired when a checkout session is completed successfully
</Card>

<Card title="payment.succeeded" icon="check">

<Card title="payment.failed" icon="x-circle">
