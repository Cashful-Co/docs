---
title: "Create Transfer"
sidebarTitle: "Create Transfer"
description: "Create a P2P transfer between customer balances"
api: "post /api/canary/transfers"
---

Move funds from one CustomerBalance to another within the same merchant.

## Request Body

<ParamField body="fromCustomerId" type="string" required>
  Customer ID sending funds
</ParamField>

<ParamField body="toCustomerId" type="string" required>
  Customer ID receiving funds
</ParamField>

<ParamField body="amount" type="number" required>
  Amount in the smallest currency unit
</ParamField>

<ParamField body="currency" type="string" required>
  Three-letter ISO currency code
</ParamField>

<ParamField body="description" type="string">
  Optional description
</ParamField>

## Responses

<ResponseField status="201" name="Transfer Created">
  <ResponseField name="id" type="string">Transfer ID</ResponseField>
  <ResponseField name="fromCustomerId" type="string">Sender ID</ResponseField>
  <ResponseField name="toCustomerId" type="string">Recipient ID</ResponseField>
  <ResponseField name="amount" type="number">Amount transferred</ResponseField>
  <ResponseField name="currency" type="string">Currency code</ResponseField>
  <ResponseField name="status" type="string">Transfer status</ResponseField>
  <ResponseField name="createdAt" type="string">Created timestamp</ResponseField>
</ResponseField>

<ResponseField status="400" name="Bad Request">
  Invalid input or validation error
</ResponseField>

<ResponseField status="401" name="Unauthorized">
  Invalid authentication
</ResponseField>

<ResponseField status="402" name="Insufficient Balance">
  Sender has insufficient balance
</ResponseField>

## Authentication

This endpoint requires Bearer token authentication.
